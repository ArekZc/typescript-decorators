Service in ES6


import angular from "angular";

class ExampleService {
    constructor(dep1, dep2) {
        this.dep1 = dep1;
        this.dep2 = dep2;
    }

    method() {
        return this.dep1 + this.dep2;
    }
}

ExampleService.$inject = ['dep1', 'dep2'];

angular.service('exampleService', ExampleService);
