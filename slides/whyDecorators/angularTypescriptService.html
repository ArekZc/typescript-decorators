Service in Typescript


import {Inject, Service} from "../decorators";

@Inject('dep1', 'dep2')
@Service({
    name: 'exampleService'
})
class ExampleService {
    private dep1: string;
    private dep2: string;

    constructor(dep1, dep2) {
        this.dep1 = dep1;
        this.dep2 = dep2;
    }

    method(): string {
        return this.dep1 + this.dep2;
    }
}
